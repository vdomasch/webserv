<script>
function sendPost() {
const fileInput = document.getElementById("myFilePicker");
const file = fileInput.files[0];
if (!file) {
document.getElementById("uploadResult").innerText = "❌ Please select a file.";
return;
}
const formData = new FormData();
formData.append("myFilePicker", file);

fetch("/cgi-bin/file_upload.py", {
method: "POST",
body: formData
})
.then(response => {
return response.text().then(text => {
if (response.ok) {
document.getElementById("uploadResult").innerText = "✅ File send";
} else {
document.getElementById("uploadResult").innerText = "❌ Error ${response.status} : ${text}";
}
fileInput.value = "";
});
})
.catch(error => {
document.getElementById("uploadResult").innerText = "❌ Error network : " + error;
});
}
</script>
